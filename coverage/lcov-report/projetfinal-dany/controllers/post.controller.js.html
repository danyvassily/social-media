module.exports.createPost = async (req, res) => {
  let fileName = ""; // Nom du fichier image
 
  if (req.file) { // Si un fichier est uploadé
    try {
      if (!["image/jpg", "image/png", "image/jpeg"].includes(req.file.mimetype)) {
        throw Error("invalid file"); // Vérification du type de fichier
      }
 
      if (req.file.size > 500000) throw Error("max size"); // Vérification de la taille du fichier
 
      fileName = `${req.body.posterId}_${Date.now()}.jpg`; // Création d'un nom de fichier unique
 
      // Utilisation du buffer pour écrire le fichier directement
      const uploadPath = './client/public/uploads/posts';
      const filePath = `${uploadPath}/${fileName}`;
      
      // Vérification si le répertoire existe
      if (!fs.existsSync(uploadPath)) {
        fs.mkdirSync(uploadPath, { recursive: true }); // Création du répertoire si non existant
      }
 
      // Écriture du fichier depuis le buffer
      fs.writeFileSync(filePath, req.file.buffer);
    } catch (err) {
      console.error("Erreur lors de l'upload:", err); // Journalisation de l'erreur
      const errors = uploadErrors(err); // Gestion des erreurs spécifiques à l'upload
      return res.status(400).json({ errors }); // Envoi des erreurs au client
    }
  }
 
  // Création d'un nouvel objet Post
  const newPost = new PostModel({
    posterId: req.body.posterId, // ID de l'utilisateur qui crée le post
    message: req.body.message, // Message du post
    picture: fileName ? `/uploads/posts/${fileName}` : "", // Chemin de l'image si existante
    comments: [], // Initialisation des commentaires
  });
 
  try {
    const post = await newPost.save(); // Sauvegarde du post dans la base de données
